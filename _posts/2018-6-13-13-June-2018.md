#### FizzBuzz
```javascript
for(let i = 1; i <= 100; i++) {
  let num = i;

  if(num % 3 === 0 && num % 5 === 0 ){
    console.log(`${num} is FizzBuzz`);
  } else if (num % 3 === 0) {
    console.log(`${num} is Fizz`);
  } else if(num % 5 === 0) {
    console.log(`${num} is Buzz`);
  } else {
    console.log(`${num}`);
  }
}
```
